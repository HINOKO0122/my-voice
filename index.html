// ... (HTML部分は同じでOK、scriptタグの中身を以下に差し替え) let audioContext; let microphone; const startBtn = document.getElementById('startBtn'); const statusText = document.getElementById('status'); startBtn.onclick = async () => { if (audioContext) return; try { // 途切れ対策1: サンプリングレートを固定し、レイテンシ（遅延）の設定を調整 const stream = await navigator.mediaDevices.getUserMedia({ audio: { echoCancellation: true, // ハウリング防止 noiseSuppression: false, // 途切れの原因になりやすいのでOFFにしてみる autoGainControl: true, latency: 0.02 // わずかなバッファを持たせて安定させる } }); // 途切れ対策2: LatencyHintを 'interactive' (低遅延重視) に設定 audioContext = new (window.AudioContext || window.webkitAudioContext)({ latencyHint: 'interactive', sampleRate: 44100 // 標準的な音質に固定 }); microphone = audioContext.createMediaStreamSource(stream); // 出力へ接続 microphone.connect(audioContext.destination); statusText.innerText = "🎤 モニター中"; startBtn.disabled = true; startBtn.style.background = "#28a745"; } catch (err) { console.error(err); statusText.innerText = "エラーが発生しました。"; } };
